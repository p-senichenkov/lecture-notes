\chapter{Равномерное приближение}

\section{Многочлен наилучшего равномерного приближения}

\begin{problem}
	Пусть $ f \in \mathcal C[a, b] $.
	Найти $ E_n(f) = \in\limits_{P_n} \|f - P_n\|_{\mathcal C[a, b]} $.

	$ E_n(f) $ называется \emph{наилучшим равномерным приближением функции} $ f $ \emph{алгебраическими многочленами степени не выше} $ n $.
\end{problem}
